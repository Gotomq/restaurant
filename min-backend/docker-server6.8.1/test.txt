#Pull server & studio:
 docker pull datastax/dse-server:6.8.1
 docker pull datastax/dse-studio

#Start a container with the DSE image using the following command
 docker run -e DS_LICENSE=accept -e JVM_EXTRA_OPTS="-Xms2g" --name my-dse -p 9042:9042 -d datastax/dse-server:6.8.1

#Start studio
 docker run -e DS_LICENSE=accept --link my-dse --name my-studio -p 9091:9091 -d datastax/dse-studio

http://localhost:9091/
